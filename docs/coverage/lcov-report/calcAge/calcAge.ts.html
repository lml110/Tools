import dayjs from 'dayjs'
import _isDate from '../_isDate/_isDate'
 
interface AgeResult {
  year: number
  month: number
  day: number
  monthTotal: number
  dayTotal: number
}
 
/**
 * @typedef {Object} AgeResult
 * @property {number} year - 相差年份
 * @property {number} month - 相差月份
 * @property {number} day - 相差天数
 * @property {number} monthTotal - 相差总月份
 * @property {number} dayTotal - 相差总天数
 */
 
/**
 * 根据出生日期计算年龄
 * @function calcAge
 * @returns {AgeResult} 年龄对象
 * @param {Date} birthDate - 出生日期
 * @param {Date} [nowDate=new Date()] - 可选 - 计算目标, 默认当前日期
 * @author 薛博仁 <boren.xue@winbaoxian.com>
 * @example
 * const age = calcAge(new Date('2000/06/06')).year
 *
 * const ageObj: AgeResult = calcAge(new Date('2000/06/06'), new Date('2019/06/06'))
 */
export default function calcAge(
  birthDate: Date,
  nowDate: Date = new Date(),
): AgeResult {
  if (!_isDate(birthDate)) throw new Error('birthDate 参数必须是日期对象')
  if (!_isDate(nowDate)) throw new Error('nowDate 参数必须是日期对象')
 
  let start = birthDate
  let end = nowDate
  if (start > end) [start, end] = [end, start]
 
  const startDay = dayjs(start)
  const endDay = dayjs(end)
 
  const diffYear = endDay.diff(startDay, 'year')
  const diffMonth = endDay.diff(startDay.add(diffYear, 'year'), 'month')
  const diffDay = endDay.diff(
    startDay.add(diffYear, 'year').add(diffMonth, 'month'),
    'day',
  )
  const diffMonthTotal = endDay.diff(startDay, 'month')
  const diffDayTotal = endDay.diff(startDay, 'day')
 
  return {
    year: diffYear,
    month: diffMonth,
    day: diffDay,
    monthTotal: diffMonthTotal,
    dayTotal: diffDayTotal,
  }
}
