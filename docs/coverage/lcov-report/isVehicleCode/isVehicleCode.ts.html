/**
 * 是否是合法的车牌号码 - 支持小汽车，货车和新能源汽车三类
 * 相关规则参考：http://www.360doc.com/content/19/1018/10/48933397_867582489.shtml
 * @function isVehicleCode
 * @returns {boolean} 是否是合法的车牌号码
 * @param {string} code - 车牌号码字符串
 * @param {boolean} isTrim - 是否需要处理空格 true-处理 false-不处理 默认不处理false
 * @throws {TypeError} isVehicleCode期望参数车牌号code必须为string类型
 * @author  watson <hangbo.s@winbaoxian.com>
 * @example // returns true
 * isVehicleCode('浙A123AA')
 */
export default function isVehicleCode(code: string, isTrim = false): boolean {
  if (typeof code !== 'string') {
    throw new TypeError('isVehicleCode期望参数车牌号code必须为string类型')
  }
  const creg = /^[京津沪渝冀豫云辽黑湘皖鲁新苏浙赣鄂桂甘晋蒙陕吉闽贵粤青藏川宁琼使领A-Z]{1}[A-Z]{1}[A-HJ-NP-Z0-9]{4}[A-HJ-NP-Z0-9挂学警港澳]{1}$/
  const xreg = /^[京津沪渝冀豫云辽黑湘皖鲁新苏浙赣鄂桂甘晋蒙陕吉闽贵粤青藏川宁琼使领A-Z]{1}[A-Z]{1}(([0-9]{5}[DF]$)|([DF][A-HJ-NP-Z0-9][0-9]{4}$))/
  const cCode = isTrim ? code.replace(/\s/g, '') : code
  if (cCode.length === 7) {
    return creg.test(cCode)
  }
  if (cCode.length === 8) {
    return xreg.test(cCode)
  }
  return false
}
